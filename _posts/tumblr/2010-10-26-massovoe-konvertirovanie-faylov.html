Чтобы не забыть.

#!/bin/bash
for file in `ls *.ext`
do
iconv -f cp1251 -t UTF-8 -o “new_$file” “$file”
done

и потом, собственно, создаем новую директорию, копируем туда все файлы new*.ext, выполняем

#!/bin/bash
for file in `ls`
do
mv $file ${file:4}
done
